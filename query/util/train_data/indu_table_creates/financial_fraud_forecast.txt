CREATE TABLE `financial_fraud_forecast` (

  `year` varchar(20) COLLATE utf8mb4_general_ci DEFAULT NULL COMMENT '上传的年报年份',

  `rate_of_inventory_change` varchar(50) COLLATE utf8mb4_general_ci DEFAULT NULL COMMENT '存货变化率',
  `accounts_payable_as_a_percentage` varchar(50) COLLATE utf8mb4_general_ci DEFAULT NULL COMMENT '应付款项占比',

  `free_cash_flow_rate_of_change` varchar(50) COLLATE utf8mb4_general_ci DEFAULT NULL COMMENT '自由现金流变动率',
  `money_cash_anomaly` int DEFAULT NULL COMMENT '货币现金异常',
  `proportion_of_cross_linked_transactions` int DEFAULT NULL COMMENT '交联交易占比',
  `cross_linked_transaction_change_rate` int DEFAULT NULL COMMENT '交联交易变化率',
  `gross_margin_anomaly` int DEFAULT NULL COMMENT '毛利率异常',

  `institutional_shareholding_ratio` int DEFAULT NULL COMMENT '机构持股比例',
  `board_size` int DEFAULT NULL COMMENT '董事会规模',
  `is_chang_offices` int DEFAULT NULL COMMENT '是否更换事务所',
  `audit_opinion` int DEFAULT NULL COMMENT '审计意见',
  `is_financial_frau` int DEFAULT NULL COMMENT '是否财务报告造假 造假为1,否则为0',
  PRIMARY KEY (`id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='财务造假预测'
